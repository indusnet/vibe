        class TicTacToe {

            constructor() {

                this.board = Array(9).fill(null);

                this.currentPlayer = 'X';

                this.gameActive = true;

                this.playerScore = 0;

                this.computerScore = 0;

                this.firstPlayer = 'player'; // Tracks who starts (alternates each game)

                

                this.cells = document.querySelectorAll('.cell');

                this.status = document.getElementById('status');

                this.resetBtn = document.getElementById('resetBtn');

                this.playerScoreEl = document.getElementById('playerScore');

                this.computerScoreEl = document.getElementById('computerScore');

                

                this.winningCombos = [

                    [0, 1, 2], [3, 4, 5], [6, 7, 8], // rows

                    [0, 3, 6], [1, 4, 7], [2, 5, 8], // columns

                    [0, 4, 8], [2, 4, 6] // diagonals

                ];

                

                this.init();

            }


            init() {

                this.cells.forEach(cell => {

                    cell.addEventListener('click', (e) => this.handleCellClick(e));

                });

                

                this.resetBtn.addEventListener('click', () => this.resetGame());

                

                // Computer goes first if it's their turn

                if (this.firstPlayer === 'computer') {

                    this.status.textContent = "Computer's turn";

                    setTimeout(() => this.computerMove(), 500);

                }

            }


            handleCellClick(e) {

                const cell = e.target;

                const index = parseInt(cell.dataset.index);

                

                if (!this.gameActive || this.board[index] || this.currentPlayer === 'O') {

                    return;

                }

                

                this.makeMove(index, 'X', cell);

            }


            makeMove(index, player, cell) {

                this.board[index] = player;

                cell.textContent = player;

                cell.classList.add(player.toLowerCase(), 'taken');

                

                if (this.checkWinner(player)) {

                    this.endGame(player);

                    return;

                }

                

                if (this.board.every(cell => cell !== null)) {

                    this.endGame('draw');

                    return;

                }

                

                this.currentPlayer = player === 'X' ? 'O' : 'X';

                

                if (this.currentPlayer === 'O') {

                    this.status.textContent = "Computer thinking...";

                    setTimeout(() => this.computerMove(), 600);

                } else {

                    this.status.textContent = "Your turn";

                }

            }


            computerMove() {

                if (!this.gameActive) return;

                

                const availableMoves = this.board

                    .map((cell, index) => cell === null ? index : null)

                    .filter(index => index !== null);

                

                if (availableMoves.length === 0) return;

                

                // Intelligent AI: use minimax algorithm

                const bestMove = this.findBestMove();

                const cell = this.cells[bestMove];

                

                this.makeMove(bestMove, 'O', cell);

            }


            // Minimax algorithm for unbeatable AI

            minimax(board, depth, isMaximizing) {

                // Check terminal states

                if (this.checkWinnerForBoard(board, 'O')) return 10 - depth;

                if (this.checkWinnerForBoard(board, 'X')) return depth - 10;

                if (board.every(cell => cell !== null)) return 0;


                if (isMaximizing) {

                    let bestScore = -Infinity;

                    for (let i = 0; i < 9; i++) {

                        if (board[i] === null) {

                            board[i] = 'O';

                            let score = this.minimax(board, depth + 1, false);

                            board[i] = null;

                            bestScore = Math.max(score, bestScore);

                        }

                    }

                    return bestScore;

                } else {

                    let bestScore = Infinity;

                    for (let i = 0; i < 9; i++) {

                        if (board[i] === null) {

                            board[i] = 'X';

                            let score = this.minimax(board, depth + 1, true);

                            board[i] = null;

                            bestScore = Math.min(score, bestScore);

                        }

                    }

                    return bestScore;

                }

            }


            findBestMove() {

                let bestScore = -Infinity;

                let bestMove = -1;

                

                // Create a copy of the board for simulation

                let boardCopy = [...this.board];

                

                for (let i = 0; i < 9; i++) {

                    if (boardCopy[i] === null) {

                        boardCopy[i] = 'O';

                        let score = this.minimax(boardCopy, 0, false);

                        boardCopy[i] = null;

                        

                        if (score > bestScore) {

                            bestScore = score;

                            bestMove = i;

                        }

                    }

                }

                

                return bestMove;

            }


            checkWinnerForBoard(board, player) {

                return this.winningCombos.some(combo => {

                    const [a, b, c] = combo;

                    return board[a] === player && board[b] === player && board[c] === player;

                });

            }


            checkWinner(player) {

                return this.winningCombos.some(combo => {

                    const [a, b, c] = combo;

                    if (this.board[a] === player && this.board[b] === player && this.board[c] === player) {

                        this.highlightWinningCells(combo);

                        return true;

                    }

                    return false;

                });

            }


            highlightWinningCells(combo) {

                combo.forEach(index => {

                    this.cells[index].classList.add('winning');

                });

            }


            endGame(result) {

                this.gameActive = false;

                

                if (result === 'X') {

                    this.status.textContent = '🎉 You Win! 🎉';

                    this.status.classList.add('winner');

                    this.playerScore++;
